#include <iostream>
#include <string>
#include <vector>
using namespace std;
class TrainSwapping {
  private:
    int ans;
    int num;
    vector<int >  vec;
  public:
    void initional(); 
    void readcase();
    void computering();
    void outResult();
    
};
void TrainSwapping::initional(){
  vec.clear();
  ans=0;
}
void TrainSwapping::readcase(){
  cin>>num;
  int tmp;
  for(int i=0;i<num;i++){
    cin>>tmp;
    vec.push_back(tmp);
  }
}
void TrainSwapping::computering(){
    int tmp,p;
    for(int i=0;i<num-1;i++){
      for(int j=i+1;j<num;j++){
         if(vec[i]>vec[j]){
            tmp=vec[i];
            vec[i]=vec[j];
            vec[j]=tmp;
            ans++;
         }
      }
    }
   
}
void TrainSwapping::outResult(){
  cout<<"Optimal train swapping takes "<<ans<<" swaps"<<endl;
}
int main(){
  TrainSwapping TR;
  int N;
  cin>>N;
  while(N--){
      TR.initional();
      TR.readcase();
      TR.computering();
      TR.outResult();
  }
  return 0;
}

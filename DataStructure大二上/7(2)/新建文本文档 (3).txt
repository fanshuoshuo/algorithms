#include <stdio.h>
#include <math.h>
int main()
{int t,l,a[51],i,j,sum;
 scanf("%d",&t);
 while(t--)
 {scanf("%d",&l);
  for (i=1;i<=l;i++)
  scanf("%d",&a[i]);
  sum=0;
  for (i=1;i<l;i++)
  for (j=i+1;j<=l;j++)
   if (a[i]>a[j]) {++sum;a[0]=a[i];a[i]=a[j];a[j]=a[0];}
  printf("Optimal train swapping takes %d swaps.\n",sum);
  
 }
 return 0;
}

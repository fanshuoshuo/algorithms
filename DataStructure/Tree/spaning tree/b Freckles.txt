using namespace std;
double ans=0;
const int MAX=760;
int father[20000];
void build(int n){
  for(int i=0;i<=n;i++)
     father[i]=i; 
}
int find(int x){
    if(father[x] == x){
		return x;
	}
	return ( father[x]=find(father[x]));
}
void virUnion(int x,int y){
    int xx=find(x);
    int yy=find(y);
    if(xx!=yy)  { 
       father[xx]=yy; 
    } 
    
}
int realUnion(int x,int y,double w){
    int xx = find(x);
	int yy = find(y);
	if(xx!= yy){
		father[xx] = yy;
		ans+=sqrt(w);
		return 1;
	}
} 
struct Edge{ 
    int x;
    int y;
    double w;
};
Edge edge[300000];
bool cmp(Edge a,Edge b){
	return a.w<b.w;
}
class ConnectTheCampus{
   private:
      int n;
      int M;
      int cnt;
    //  double ans; 
      double point[MAX][2];
   public :
      void initial(int );
      void readcase();
      void computering();
}; 
void ConnectTheCampus::initial(int num){
    n=num;
    ans=0;
    cnt=0; 
    memset(point,0,sizeof(point)); 
    build(n); 
} 
void ConnectTheCampus::readcase(){
    for(int i = 1; i <= n; i++){
		cin >> point[i][0] >> point[i][1];
	}

	int k=0;
	//º∆À„æ‡¿Î 
	for(int i = 1; i < n; i++){
		for(int j = i+1; j <= n;j++){
			edge[k].w = (point[i][0]-point[j][0])*(point[i][0]-point[j][0])+(point[i][1]-point[j][1])*(point[i][1]-point[j][1]);
			edge[k].x = i;
			edge[k].y = j;
			k++; 
		}
	}
	cnt=k;  
 }
void ConnectTheCampus::computering(){
     for(int i = 0; i <cnt; i++){
    // cout<<edge[i].w<<endl;  
     }
     sort(edge,edge+cnt,cmp);
     for(int i = 0; i <cnt; i++){
        // cout<<edge[i].w<<endl;  
		 realUnion(edge[i].x,edge[i].y,edge[i].w);
   	}
    	printf("%.2f\n",ans);
	
}
int main(){
    ConnectTheCampus C;
    int n;
    int t;
    cin>>t;
    
    while(t--){
      cin>>n;
      C.initial(n);
      C.readcase();
      C.computering();
      if(t!=0)
        cout<<endl;
    }
   system("pause");
    return 0;
}
